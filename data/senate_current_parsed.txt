(function () {
  const COOKIE_NAME = 'qs_bills_open';
  const COOKIE_DAYS = 180;

  const toggle = document.getElementById('qs-toggle');
  const panel  = document.getElementById('qs-slider');
  const close  = document.getElementById('qs-close');

  const openers = document.querySelectorAll('.js-open-qs-slider');

  if (!toggle || !panel) return;

  let lastFocus = null;

  function setCookie(name, value, days) {
    const d = new Date();
    d.setTime(d.getTime() + (days * 24 * 60 * 60 * 1000));
    document.cookie = name + '=' + encodeURIComponent(value) +
      '; expires=' + d.toUTCString() +
      '; path=/; SameSite=Lax';
  }

  function getCookie(name) {
    const target = name + '=';
    const parts = document.cookie.split(';');
    for (let i = 0; i < parts.length; i++) {
      const c = parts[i].trim();
      if (c.indexOf(target) === 0) return decodeURIComponent(c.substring(target.length));
    }
    return null;
  }

  function getFocusable(container) {
    return Array.from(
      container.querySelectorAll(
        'a[href], button:not([disabled]), input:not([disabled]), select:not([disabled]), textarea:not([disabled]), [tabindex]:not([tabindex="-1"])'
      )
    ).filter(el => el.offsetParent !== null);
  }

  function openPanel(userInitiated) {
    if (panel.hidden) panel.hidden = false;
    requestAnimationFrame(() => panel.classList.add('is-open'));

    toggle.setAttribute('aria-expanded', 'true');

    if (userInitiated) {
      lastFocus = document.activeElement;
      const focusables = getFocusable(panel);
      if (focusables.length) focusables[0].focus();
      else {
        panel.setAttribute('tabindex', '-1');
        panel.focus();
      }
    }

    setCookie(COOKIE_NAME, '1', COOKIE_DAYS);
    document.addEventListener('keydown', onKeyDown, true);
  }

  function closePanel() {
    panel.classList.remove('is-open');
    toggle.setAttribute('aria-expanded', 'false');
    setCookie(COOKIE_NAME, '0', COOKIE_DAYS);

    document.removeEventListener('keydown', onKeyDown, true);

    window.setTimeout(() => {
      panel.hidden = true;
      if (lastFocus && typeof lastFocus.focus === 'function') lastFocus.focus();
    }, 180);
  }

  function onKeyDown(e) {
    if (e.key === 'Escape' && !panel.hidden) {
      e.preventDefault();
      closePanel();
    }
  }

  // Toggle click (existing)
  toggle.addEventListener('click', function () {
    if (typeof window.capturetheclick === 'function') window.capturetheclick();

    const expanded = toggle.getAttribute('aria-expanded') === 'true';
    if (expanded) closePanel();
    else openPanel(true);
  });

  // Close button
  if (close) close.addEventListener('click', closePanel);

  // Additional openers
  openers.forEach(el => {
    el.addEventListener('click', function (e) {
      e.preventDefault();
      if (typeof window.capturetheclick === 'function') window.capturetheclick();
      openPanel(true);
    });
  });

  // Restore state on load
  const saved = getCookie(COOKIE_NAME);
  if (saved === '1') openPanel(false);
})();

function GetClock(){

dx = d.toGMTString();
dx = dx.substr(0,dx.length -4);
d.setTime(Date.parse(dx))
d.setHours(d.getHours() + tzOffset);
nday   = d.getDay();
nmonth = d.getMonth();
ndate  = d.getDate();
nyear = d.getYear();
if(nyear<1000) nyear=nyear+1900;
